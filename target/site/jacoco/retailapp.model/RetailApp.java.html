import org.apache.commons.lang.builder.EqualsBuilder;

@Entity
public class RetailApp implements Serializable{

	@Id

	public void setName(String name) {
		this.name = name;
	}

		return price;
	}

	public void setPrice(double price) {
		this.price = price;
	}

		return date;

	public void setDate(String date) {
